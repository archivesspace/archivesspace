def call(records)
  record_id_to_repo_ids = build_id_map(records)

  repositories = Repository.all.group_by(&:id)

  result = {}
  records.each do |record|
    result[record] = record_id_to_repo_ids
                     .fetch(record.id, [])
                     .map {|repo_ids| repositories.fetch(repo_ids)}
                     .flatten
  end

  result
end
