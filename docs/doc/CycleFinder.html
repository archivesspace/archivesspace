def each
  cycles_found = []

  @graph.keys.each do |start_node|
    @ticker.tick
    next if cycles_found.include?(start_node)

    find_cycles(start_node).each do |cycle|
      unless cycles_found.include?(cycle.node)
        cycles_found << cycle.node
        yield cycle.node
      end
    end
  end
end
