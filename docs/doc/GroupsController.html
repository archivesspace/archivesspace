def update
  params[:group][:grants_permissions] ||= []
  params[:group][:member_usernames] ||= []

  handle_crud(:instance => :group,
              :model => Accession,
              :obj => JSONModel(:group).find(params[:id]),
              :replace => false,
              :on_invalid => ->(){
                return render :action => "edit"
              },
              :on_valid => ->(id){
                redirect_to(:controller => :groups, :action => :index)
              })
end
