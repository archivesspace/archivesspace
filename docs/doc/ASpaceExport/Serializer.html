def self.with_namespace(prefix, xml)
  ns = xml.doc.root.namespace_definitions.find{|ns| ns.prefix == prefix}
  xml.instance_variable_set(:@sticky_ns, ns)
  yield 
  xml.instance_variable_set(:@sticky_ns, nil)
end
