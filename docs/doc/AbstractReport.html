def each
  DB.open do |db|
    dataset = query(db)
    dataset = scope_by_repo_id(dataset) if @repo_id
    dataset.each do |row|
      yield(Hash[headers.map { |h|
        val = (processor.has_key?(h))?processor[h].call(row):row[h.intern]
        [h, val]
      }])
    end
  end
end
