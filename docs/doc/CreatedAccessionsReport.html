def processor
  {
    'identifier' => proc {|record| ASUtils.json_parse(record[:identifier] || "[]").compact.join("-")},
    'create_date' => proc {|record| record[:create_time].strftime("%Y-%m-%d")},
    'create_time' => proc {|record| record[:create_time].strftime("%H:%M:%S")}
  }
end
