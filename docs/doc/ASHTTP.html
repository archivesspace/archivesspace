def self.start_uri(uri, opts = {})
  use_ssl = (uri.scheme == "https")
  opts = {:use_ssl => use_ssl}.merge(opts)

  Net::HTTP.start(uri.host, uri.port, opts) do |http|
    yield http
  end
end
