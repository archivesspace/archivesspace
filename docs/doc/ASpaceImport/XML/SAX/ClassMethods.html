def handler_name(path, prefix = '' )
  @sticky_nodes ||= {}
  parts = path.split("/").reverse
  handler_name = prefix 
  while parts.length > 1
    @sticky_nodes[parts.last] = true
    handler_name << "_#{parts.pop}"
  end

  handler_name << "_#{parts.pop}"
end

def ignore(path)
  with(path) {|*| @ignore = true }
  and_in_closing(path) {|*| @ignore = false }
end
