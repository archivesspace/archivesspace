def self.value(val)
  vals = val.is_a?(Array) ? val : val.split(/,/)

  result = vals.map {|elt| Integer(elt)}.uniq

  if result.length > AppConfig[:max_page_size].to_i
    raise ArgumentError.new("ID set cannot contain more than #{AppConfig[:max_page_size]}n IDs")
  end

  result
end
