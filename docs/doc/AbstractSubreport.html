def get_content
  array = []
  query.each do |result|
    row = result.to_hash
    fix_row(row)
    array.push(row)
  end
  if array.empty?
    nil
  else
    array.push(code) if ['pdf', 'html', 'rtf'].include? format
    array
  end
end
