def add_terminal_error(error, counter)
  if counter
    @terminal_error = I18n.t("bulk_import.error.stopped", row: counter, msg: error)
  else
    @terminal_error = I18n.t("bulk_import.error.initialize", msg: error)
  end
  end_row
end
