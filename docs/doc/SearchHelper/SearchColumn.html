def initialize(label, value_block, opts, search_data)
  @field = opts[:field]
  @label = label
  @value_block = value_block
  @classes = "col "
  @classes << opts[:class] if opts[:class]
  @sortable = opts[:sortable] || false
  @sort_by = opts[:sort_by] || ""
  @search_data = search_data
end
