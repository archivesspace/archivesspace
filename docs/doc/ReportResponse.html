Overview

this is a generic wrapper for reports reponses. JasperReports do not need a reponse wrapper and can return reports on formats using the to_FORMAT convention. “Classic” AS reports need a wrapper to render the report in a specific format.

def generate
  if  @report.is_a?(JasperReport) 
    format = @report.format    
    String.from_java_bytes( @report.render(format.to_sym, @params) ) 
  else
    file = File.join( File.dirname(__FILE__), "../../views/reports/report.erb")
    @params[:html_report] ||= proc {  ERB.new( IO.read( file )).result(@report.get_binding) }
  
    format = @report.format

    klass = Object.const_get("#{format.upcase}Response")
    klass.send(:new, @report, @params).generate
  end
end
