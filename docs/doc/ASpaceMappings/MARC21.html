def self.get_aspace_source_code(code)
  case code.to_s
  when '0'; 'lcsh'
  when '1'; 'lcshac'
  when '2'; 'mesh'
  when '3'; 'nal'
  when '4'; 'ingest'
  when '5'; 'cash'
  when '6'; 'rvm'
  else; nil
  end
end

def self.get_marc_source_code(code)

  marc_code = case code
              when 'naf', 'lcsh', 'lcnaf'; 0
              when 'lcshac'; 1
              when 'mesh'; 2
              when 'nal'; 3
              when nil; 4
              when 'cash'; 5
              when 'rvm'; 6
              else; 7
              end

  marc_code.to_s
end
