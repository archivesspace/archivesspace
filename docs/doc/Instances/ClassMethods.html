def sequel_to_jsonmodel(objs, opts = {})
  jsons = super

  jsons.each do |json|
    if json['instances']
      json['instances'].reject! {|instance|
        instance['instance_type'] == 'digital_object' && !instance['digital_object']
      }
    end
  end

  jsons
end
