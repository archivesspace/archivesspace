def validate_mixed_content_field(field = :title)
  value = begin
    respond_to?(field) ? send(field) : self[field]
  rescue
    self[field]
  end

  return if value.nil? || value.to_s.strip.empty?

  if (msg = MixedContentValidator.error_for_inline_ead(value))
    errors.add(field, msg)
  end
end
