def self.sequel_to_jsonmodel(obj, opts = {})
  json = super

  json["linked_instances"] = []

  obj.linked_records(:instance_do_link).each do |link|
    json["linked_instances"].push({
        "ref" => link.resource ? link.resource.uri : link.archival_object.uri 
    })
  end

  json
end
