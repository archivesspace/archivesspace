def mix(hash1, hash2, hash3=nil)
  if hash3
    hash2 = mix(hash2, hash3)
  end
  hash1.merge(hash2) do |key, one, two|
    if one.is_a?(Hash) && two.is_a?(Hash)
      mix(one, two)
    elsif one.is_a?(Proc) && two.is_a?(Proc)
      [one, two]
    elsif one.is_a?(Array) && two.is_a?(Proc)
      one << two
    else
      two
    end
  end
end
