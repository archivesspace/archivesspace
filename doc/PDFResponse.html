def generate
  
  estimated_pdf_length = @html_report.length
  output = java.io.ByteArrayOutputStream.new(estimated_pdf_length)
  begin
    dom = java_dom(@html_report)
    render_pdf(dom, output)
  ensure
    output.close
  end
end
