def self.blobField(name, opts = {})
  if $db_type == :postgres
    [name, :bytea, opts]
  elsif $db_type == :h2
    [name, String, opts.merge(:size => 128000)]
  else
    [name, :blob, opts]
  end
end

def self.mediumBlobField(name, opts = {})
  if $db_type == :postgres
    [name, :bytea, opts]
  elsif $db_type == :h2
    [name, String, opts.merge(:size => 128000)]
  elsif $db_type == :mysql
    [name, :mediumblob, opts]
  else
    [name, :blob, opts]
  end
end

def self.textBlobField(name, opts = {})

  else
