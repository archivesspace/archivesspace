def generate_label_rows(objects)

  rows = []
  
  objects.each do |obj|

    instances = obj.instances
    instances.each do |i|
      c = i['container']
      next unless c
      crow = [] 
      if c['type_1'] && c['indicator_1'] 
        crow << "#{c['type_1']} #{c['indicator_1']}"
      end
      if c['barcode_1']
        crow << c['barcode_1']
      end
      rows << crow
    end
    rows.push(*generate_label_rows(obj.children))

  end
  
  rows
end
