def entity_gsub!(chars)
  mapping = {
    '&' => '&amp;',
    '<' => '&lt;',
    '>' => '&gt;',
    '"' => '&quot;',
    "'" => '&apos;'
  }
  re = /&(?!amp;)|[<>'"]/
  chars.gsub!(re, mapping)
  chars
end

def start_element_namespace(name, attrs=[], prefix=nil, uri=nil, ns=[])
  return if name == 'ridiculous_wrapper_element'
  @file << "<#{name}"
  unless attrs.empty?
    attrs.each do |attr|
      @file << " #{attr.localname}=\"#{entity_gsub!(attr.value)}\""
    end
  end
  @file << '>'
end
