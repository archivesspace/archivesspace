def add_objects(model, *ids)
  flat_ids = ids.flatten
  return if flat_ids.empty?
  @graph[model] ||= Set.new
  @graph[model].merge(flat_ids)

  self
end

def each(&block)
  if block_given?
    @graph.keys.each do |model|
      block.call(model, @graph[model].to_a)
    end
  else
    Enumerator.new do |y|
      @graph.map {|model, ids|
        y << [model, ids.to_a]
      }
    end
  end
end
