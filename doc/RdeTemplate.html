def self.sequel_to_jsonmodel(objs, opts = {})
  jsons = super

  jsons.zip(objs).each do |json, obj|
    json['order'] = ASUtils.json_parse(obj.order)
    json['visible'] = ASUtils.json_parse(obj.visible)
    json['defaults'] = ASUtils.json_parse(obj.defaults)
  end

  jsons
end
