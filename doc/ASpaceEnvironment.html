def self.init(environment = :auto)
  return if @environment      # Already initialised

  if environment != :auto
    @environment = environment
  elsif ENV["ASPACE_INTEGRATION"] == "true"
    @environment = :integration
  else
    @environment = :production
  end

  prepare_database
end
