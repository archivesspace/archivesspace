def self.extended(base)
  blob_columns = base.db_schema.select {|column, defn| defn[:type] == :blob}.keys

  base.instance_eval do
    @blob_columns_to_fix = (!blob_columns.empty? && DB.needs_blob_hack?) ? Array(blob_columns) : []
  end
end
